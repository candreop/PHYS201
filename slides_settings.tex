% Blocks:
%  001      - white bg with baby blue header bg, black fg
%  minirem  - mini reminder in a main slide
%  que      - question block in main slides
%  ans      - answer block in main slides
%  exmplque - question block for worked example slides
%  exmplans - question block for worked example slides
%

% 001
\newenvironment<>{block001}[1]{%
\begin{actionenv}#2%
  \definecolor{block001_header}{RGB}{204,229,255}
  \def\insertblocktitle{#1}%
  \par%
  \mode<presentation>{%
    \setbeamercolor{block title}{fg=black,bg=block001_header}
    \setbeamercolor{block body}{fg=black,bg=white}
    \setbeamercolor{itemize item}{fg=orange!20!black}
    \setbeamertemplate{itemize item}[triangle]
  }%
  \usebeamertemplate{block begin}}
  {\par\usebeamertemplate{block end}
\end{actionenv}}
